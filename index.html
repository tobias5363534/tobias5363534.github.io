<script>
// Función para obtener la hora actual según la zona horaria seleccionada
Function obtenerHora() {
Const pais = document.getElementById(“pais”).value;
Const formatoHora = document.getElementById(“formatoHora”).value;
Const fechaHora = new Date().toLocaleString(“es-ES”, { timeZone: pais,
hour12: (formatoHora === “12”) });
Document.getElementById(“date-time”).textContent = `La hora en
${pais} es: ${fechaHora}`;
}
// Llama a la función obtenerHora() cuando se carga la página
Document.addEventListener(“DOMContentLoaded”, obtenerHora);
// Actualizar la hora cada segundo
setInterval(obtenerHora, 1000);
